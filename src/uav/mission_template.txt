######### MISSION TEMPLATE - GPS-DENIED RC OVERRIDE PATTERN
# use this as a starting point for any new drone missions
# this pattern is confirmed to work for arming and flight on our cube orange

from pymavlink import mavutil # verified flight library
import time # for timing and sleeps
import v2v_bridge # our custom radio bridge talker

################################# config stuff
# basic settings i usually set
CONNECTION_STRING = "/dev/ttyACM0"   # drone wire (usb/serial)
BAUD_RATE = 57600                    # confirmed working baud rate
ESP32_PORT = "/dev/ttyUSB0"          # the radio bridge hardware

# mission params
TARGET_ALT = 1.3  # hover height to pass rule (4.2ft)

# throttle settings i tuned
THROTTLE_MIN = 1000   # motors completely off
THROTTLE_IDLE = 1150  # motors spinning but no lift
THROTTLE_CLIMB = 1650 # power needed to lift off
THROTTLE_HOVER = 1500 # rough middle power to hold height

############################ the mavlink helpers

def set_throttle(master, pwm): # rc override heartbeat for channel 3
    # ardupilot needs this pulse frequently or it cuts the motors
    master.mav.rc_channels_override_send(master.target_system, master.target_component, 0, 0, pwm, 0, 0, 0, 0, 0)

def get_lidar_alt(master): # checks real distance from the floor
    # grabbing the raw distance packet from the lidarlitev3 sensor
    msg = master.recv_match(type='DISTANCE_SENSOR', blocking=True, timeout=1.0)
    if msg: return msg.current_distance / 100.0 # convert cm to meters
    return 0.0 # return 0 if the sensor is ghosting

def change_mode(master, mode: str): # changes drone flight mode
    mapping = master.mode_mapping()
    if mode not in mapping: return
    mode_id = mapping[mode]
    master.mav.set_mode_send(master.target_system, mavutil.mavlink.MAV_MODE_FLAG_CUSTOM_MODE_ENABLED, mode_id)
    time.sleep(1) # wait for mode shift

#################### main logic skeleton

def main(): # main mission boss
    # 1. connect wire and radio
    master = mavutil.mavlink_connection(CONNECTION_STRING, baud=BAUD_RATE)
    master.wait_heartbeat() # wait for pulse
    
    bridge = v2v_bridge.V2VBridge(ESP32_PORT, name="Drone-Template")
    
    try:
        bridge.connect() # open radio
        
        # 2. auto launch sequence
        change_mode(master, "STABILIZE") # switch to stabilize for manual override
        
        # arming motors
        master.mav.command_long_send(master.target_system, master.target_component, mavutil.mavlink.MAV_CMD_COMPONENT_ARM_DISARM, 0, 1, 0,0,0,0,0,0)
        time.sleep(2) # wait for spinup
        
        # climb phase
        while True: # climb until high enough
            alt = get_lidar_alt(master)
            if alt >= TARGET_ALT: break
            set_throttle(master, THROTTLE_CLIMB) # pushing up
            time.sleep(0.1) # quick loop for heartbeat
        
        # 3. MISSION BODY
        while True: # the actual mission work goes here
            set_throttle(master, THROTTLE_HOVER) # ALWAYS send heartbeat in every loop
            
            # example: wait for UGV or track target
            # if mission_done(): break
            
            time.sleep(0.1) # keep it smooth

        # 4. shutdown / landing
        print("Mission Complete. Landing safely...")
        change_mode(master, "LAND") # switch to land mode
        set_throttle(master, 0) # release control to autopilot
        
        while True: # wait for disarm
            msg = master.recv_match(type='HEARTBEAT', blocking=False)
            if msg and not (msg.base_mode & mavutil.mavlink.MAV_MODE_FLAG_SAFETY_ARMED):
                print("Landed and disarmed safely.")
                break
            time.sleep(0.5)

    except KeyboardInterrupt: # e-stop / user abort
        print("User Abort. Forcing Landing...")
        change_mode(master, "LAND")
        set_throttle(master, 0)
        time.sleep(1)
    finally:
        bridge.stop()
        set_throttle(master, 0) # final safety

if __name__ == "__main__":
    main()
